#include <stdio.h>


#define BOARDSIZE 7
#define WALLS 10

// Représentations des joueurs
char board[BOARDSIZE][BOARDSIZE];
int player1_x = 0, player1_y = 3;
int player2_x = 6, player2_y = 3;
int walls_player1 = WALLS, walls_player2 = WALLS;

// Initialisation du plateau
void initialize_board() {
    for (int i = 0; i < BOARDSIZE; i++) {
        for (int j = 0; j < BOARDSIZE; j++) {
            board[i][j] = ' ';
        }
    }
    board[player1_x][player1_y] = '1'; // Position de joueur 1
    board[player2_x][player2_y] = '2'; // Position de joueur 2
}

// Affichage du plateau
void show_board() {
    for (int i = 0; i < BOARDSIZE; i++) {
        for (int j = 0; j < BOARDSIZE; j++) {
            printf("%c ", board[i][j]);
        }
        printf("\n");
    }
}

// Déplacement des joueurs
void play_move(int player, int x, int y) {
    if (x >= 0 && x < BOARDSIZE && y >= 0 && y < BOARDSIZE && board[x][y] == ' ') {
        if (player == 1) {
            board[player1_x][player1_y] = ' ';
            player1_x = x;
            player1_y = y;
            board[x][y] = '1';
        } else if (player == 2) {
            board[player2_x][player2_y] = ' ';
            player2_x = x;
            player2_y = y;
            board[x][y] = '2';
        }
        printf("Player %d moved to (%d, %d)\n", player, x, y);
    } else {
        printf("Invalid move!\n");
    }
}

// Placement des murs
void play_wall(int player, int x, int y) {
    if (player == 1 && walls_player1 > 0 && board[x][y] == ' ') {
        board[x][y] = '|';
        walls_player1--;
        printf("Player 1 placed a wall at (%d, %d)\n", x, y);
    } else if (player == 2 && walls_player2 > 0 && board[x][y] == ' ') {
        board[x][y] = '|';
        walls_player2--;
        printf("Player 2 placed a wall at (%d, %d)\n", x, y);
    } else {
        printf("Cannot place wall!\n");
    }
}

// Vérification du gagnant
int check_winner() {
    if (player1_x == BOARDSIZE - 1) {
        printf("Player 1 wins!\n");
        return 1;
    } else if (player2_x == 0) {
        printf("Player 2 wins!\n");
        return 2;
    }
    return 0;
}

int main() {
    initialize_board();
    int winner = 0;
    while (!winner) {
        show_board();
        int player, x, y;
        printf("Enter move (player x y): ");
        scanf("%d %d %d", &player, &x, &y);

        if (player == 1 || player == 2) {
            play_move(player, x, y);
            winner = check_winner();
        } else {
            printf("Invalid player!\n");
        }
    }
    return 0;
}
